001    package org.javasimon.spring;
002    
003    import java.io.Serializable;
004    
005    import org.javasimon.Manager;
006    import org.javasimon.SimonManager;
007    import org.javasimon.Split;
008    import org.javasimon.source.StopwatchSource;
009    
010    import org.aopalliance.intercept.MethodInterceptor;
011    import org.aopalliance.intercept.MethodInvocation;
012    
013    /**
014     * Basic method interceptor that measures the duration of the intercepted call with a Stopwatch.
015     * Class can be overridden in case more sophisticated measuring needs to be provided - this all should
016     * happen in {@link #processInvoke(org.aopalliance.intercept.MethodInvocation, org.javasimon.Split)} method.
017     *
018     * @author Erik van Oosten
019     */
020    public class BasicMonitoringInterceptor implements MethodInterceptor, Serializable {
021            private final StopwatchSource<MethodInvocation> stopwatchSource;
022    
023            /**
024             * Constructor with specified {@link org.javasimon.source.MonitorSource}.
025             *
026             * @param stopwatchSource stopwatch provider for method invocation
027             */
028            public BasicMonitoringInterceptor(StopwatchSource<MethodInvocation> stopwatchSource) {
029                    this.stopwatchSource = stopwatchSource;
030            }
031    
032            /**
033             * Constructor with specified {@link org.javasimon.Manager}.
034             */
035            public BasicMonitoringInterceptor(Manager manager) {
036                    this(new SpringStopwatchSource(manager).cache());
037            }
038    
039            /**
040             * Default constructor using {@link org.javasimon.SimonManager#manager}.
041             */
042            public BasicMonitoringInterceptor() {
043                    this(new SpringStopwatchSource(SimonManager.manager()).cache());
044            }
045    
046            /**
047             * Performs method invocation and wraps it with Stopwatch.
048             *
049             * @param invocation method invocation
050             * @return return object from the method
051             * @throws Throwable anything thrown by the method
052             */
053            public final Object invoke(MethodInvocation invocation) throws Throwable {
054                    final Split split = stopwatchSource.start(invocation);
055                    try {
056                            return processInvoke(invocation, split);
057                    } finally {
058                            split.stop();
059                    }
060            }
061    
062            /**
063             * Method with default invoke (just calls proceed). It can be overridden and overriding method can stop
064             * the provided split. If split is not stopped it will be stopped right after this method finishes.
065             * Any caught throwable should be rethrown again. Method should call {@code invocation.proceed()}.
066             *
067             * @param invocation method invocation
068             * @param split running split for this monitored action
069             * @return return object from the method
070             * @throws Throwable
071             */
072            protected Object processInvoke(MethodInvocation invocation, @SuppressWarnings("UnusedParameters") Split split) throws Throwable {
073                    return invocation.proceed();
074            }
075    }

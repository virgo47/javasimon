019    /**
020     * SimonUtils provides static utility methods.
021     * <p/>
022     * <h3>Human readable outputs</h3>
023     * Both {@link org.javasimon.Stopwatch} and {@link org.javasimon.Counter} provide human readable
024     * {@code toString} outputs. All nanosecond values are converted into few valid digits with
025     * proper unit (ns, us, ms, s) - this is done via method {@link #presentNanoTime(long)}.
026     * Max/min counter values are checked for undefined state (max/min long value is converted
027     * to string "undef") - via method {@link #presentMinMaxCount(long)}.
028     * <p/>
029     * <h3>Aggregation utilities</h3>
030     * It is possible to sum up (aggregate) values for a subtree for a particular Simon type using
031     * {@link #calculateCounterAggregate(org.javasimon.Simon)} or {@link #calculateStopwatchAggregate(org.javasimon.Simon)}.
032     * Methods come also in versions allowing to filter by {@link org.javasimon.SimonFilter}.
033     * <p/>
034     * <h3>Simon tree operations</h3>
035     * For various debug purposes there is a method that creates string displaying the whole Simon sub-tree.
036     * Here is example code that initializes two random Simons and prints the whole Simon hierarchy
037     * (note that the method can be used to obtain any sub-tree of the hierarchy):
038     * <pre>
039     * Split split = SimonManager.getStopwatch("com.my.other.stopwatch").start();
040     * SimonManager.getCounter("com.my.counter").setState(SimonState.DISABLED, false);
041     * split.stop();
042     * System.out.println(SimonUtils.simonTreeString(SimonManager.getRootSimon()));</pre>
043     * And the output is:
044     * <pre>
045     * (+): Unknown Simon:  [ ENABLED]
046     *   com(+): Unknown Simon:  [com INHERIT]
047     *     my(+): Unknown Simon:  [com.my INHERIT]
048     *       other(+): Unknown Simon:  [com.my.other INHERIT]
049     *         stopwatch(+): Simon Stopwatch: total 24.2 ms, counter 1, max 24.2 ms, min 24.2 ms, mean 24.2 ms [com.my.other.stopwatch INHERIT]
050     *       counter(-): Simon Counter: counter=0, max=undef, min=undef [com.my.counter DISABLED]</pre>
051     * Notice +/- signs in parenthesis that displays effective Simon state (enabled/disabled), further
052     * details are printed via each Simon's {@code toString} method.
053     *
054     * <h3>Other utilities</h3>
055     * It is possible to obtain "local name" of the Simon (behind the last dot) via {@link #localName(String)}
056     * or check if the name is valid Simon name via {@link #checkName(String)}.
057     *
058     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
059     * @author Radovan Sninsky
060     * @since 1.0
061     */

125            /**
126             * Returns nano-time in human readable form with unit. Number is always from 10 to 9999
127             * except for seconds that are the biggest unit used. In case of undefined value ({@link Long#MAX_VALUE} or
128             * {@link Long#MIN_VALUE}) string {@link #UNDEF_STRING} is returned, so it can be used for min/max values as well.
129             *
130             * @param nanos time in nanoseconds
131             * @return human readable time string
132             */
133            public static String presentNanoTime(long nanos) {
134                    if (nanos == Long.MAX_VALUE || nanos == Long.MIN_VALUE) {
135                            return UNDEF_STRING;
136                    }
137                    return presentNanoTimePrivate((double) nanos);
138            }
139    
140            /**
141             * Returns nano-time in human readable form with unit. Number is always from 10 to 9999
142             * except for seconds that are the biggest unit used.
143             *
144             * @param nanos time in nanoseconds
145             * @return human readable time string
146             */
147            public static String presentNanoTime(double nanos) {
148                    if (nanos == Double.MAX_VALUE) {
149                            return UNDEF_STRING;
150                    }
151                    return presentNanoTimePrivate(nanos);
152            }
153    
154            private static String presentNanoTimePrivate(double time) {
155                    if (Math.abs(time) < 1d) {
156                            return "0";
157                    }
158    
159                    if (time < UNIT_PREFIX_FACTOR) {
160                            return ((long) time) + " ns";
161                    }
162    
163                    time /= UNIT_PREFIX_FACTOR;
164                    if (time < UNIT_PREFIX_FACTOR) {
165                            return formatTime(time, " us");
166                    }
167    
168                    time /= UNIT_PREFIX_FACTOR;
169                    if (time < UNIT_PREFIX_FACTOR) {
170                            return formatTime(time, " ms");
171                    }
172    
173                    time /= UNIT_PREFIX_FACTOR;
174                    return formatTime(time, " s");
175            }
176    
177            private static synchronized String formatTime(double time, String unit) {
178                    if (time < TEN) {
179                            return UNDER_TEN_FORMAT.format(time) + unit;
180                    }
181                    if (time < HUNDRED) {
182                            return UNDER_HUNDRED_FORMAT.format(time) + unit;
183                    }
184                    return DEFAULT_FORMAT.format(time) + unit;
185            }

187            /**
188             * Returns timestamp in human readable (yet condensed) form "yyMMdd-HHmmss.SSS".
189             *
190             * @param timestamp timestamp in millis
191             * @return timestamp as a human readable string
192             */
193            public static String presentTimestamp(long timestamp) {
194                    if (timestamp == 0) {
195                            return UNDEF_STRING;
196                    }
197                    synchronized (TIMESTAMP_FORMAT) {
198                            return TIMESTAMP_FORMAT.format(new Date(timestamp));
199                    }
200            }

232            /**
233             * Returns multi-line string containing Simon tree starting with the specified Simon.
234             * Root Simon can be used to obtain tree with all Simons. Returns {@code null} for
235             * input value of null or for NullSimon or any Simon with name equal to null (anonymous
236             * Simons) - this is also the case when the Manager is disabled and tree for its root
237             * Simon is requested.
238             *
239             * @param simon root Simon of the output tree
240             * @return string containing the tree or null if the Simon is null Simon
241             */
242            public static String simonTreeString(Simon simon) {
243                    if (simon == null || simon.getName() == null) {
244                            return null;
245                    }
246                    StringBuilder sb = new StringBuilder();
247                    printSimonTree(0, simon, sb);
248                    return sb.toString();
249            }
250    
251            private static void printSimonTree(int level, Simon simon, StringBuilder sb) {
252                    printSimon(level, simon, sb);
253                    for (Simon child : simon.getChildren()) {
254                            printSimonTree(level + 1, child, sb);
255                    }
256            }
257    
258            private static void printSimon(int level, Simon simon, StringBuilder sb) {
259                    for (int i = 0; i < level; i++) {
260                            sb.append("  ");
261                    }
262                    sb.append(localName(simon.getName()))
263                            .append('(')
264                            .append(simon.isEnabled() ? '+' : '-')
265                            .append("): ")
266                            .append(simon.toString())
267                            .append('\n');
268            }

270            /**
271             * Returns last part of Simon name - local name.
272             *
273             * @param name full Simon name
274             * @return string containing local name
275             */
276            public static String localName(String name) {
277                    int ix = name.lastIndexOf(Manager.HIERARCHY_DELIMITER);
278                    if (ix == -1) {
279                            return name;
280                    }
281                    return name.substring(ix + 1);
282            }

338            // method is extracted, so the stack trace index is always right
339            private static String generatePrivate(String suffix, boolean includeMethodName) {
340                    StackTraceElement stackElement = Thread.currentThread().getStackTrace()[CLIENT_CODE_STACK_INDEX];
341                    StringBuilder nameBuilder = new StringBuilder(stackElement.getClassName());
342                    if (includeMethodName) {
343                            nameBuilder.append('.').append(stackElement.getMethodName());
344                    }
345                    if (suffix != null) {
346                            nameBuilder.append(suffix);
347                    }
348                    return nameBuilder.toString();
349            }

361            /**
362             * Calls a block of code with stopwatch around and returns result.
363             *
364             * @param name name of the Stopwatch
365             * @param callable callable block of code
366             * @param <T> return type
367             * @return whatever block of code returns
368             * @throws Exception whatever block of code throws
369             * @since 3.0
370             */
371            public static <T> T doWithStopwatch(String name, Callable<T> callable) throws Exception {
372                    Split split = SimonManager.getStopwatch(name).start();
373                    try {
374                            return callable.call();
375                    } finally {
376                            split.stop();
377                    }
378            }
379    
380            /**
381             * Calls a block of code with stopwatch around, can not return any result or throw an exception
382             * (use {@link #doWithStopwatch(String, java.util.concurrent.Callable)} instead).
383             *
384             * @param name name of the Stopwatch
385             * @param runnable wrapped block of code
386             * @since 3.0
387             */
388            public static void doWithStopwatch(String name, Runnable runnable) {
389                    Split split = SimonManager.getStopwatch(name).start();
390                    try {
391                            runnable.run();
392                    } finally {
393                            split.stop();
394                    }
395            }

397            private static final String SHRINKED_STRING = "...";
398    
399            /**
400             * Shrinks the middle of the input string if it is too long, so it does not exceed limitTo.
401             *
402             * @since 3.2
403             */
404            public static String compact(String input, int limitTo) {
405                    if (input == null || input.length() <= limitTo) {
406                            return input;
407                    }
408    
409                    int headLength = limitTo / 2;
410                    int tailLength = limitTo - SHRINKED_STRING.length() - headLength;
411                    if (tailLength < 0) {
412                            tailLength = 1;
413                    }
414    
415                    return input.substring(0, headLength) + SHRINKED_STRING + input.substring(input.length() - tailLength);
416            }

422             * @return aggregated statistics
423             * @since 3.5
424             */
425            public static StopwatchAggregate calculateStopwatchAggregate(Simon simon) {
426                    return calculateStopwatchAggregate(simon, null);
427            }

429            /**
430             * Aggregate statistics from all stopwatches in hierarchy that pass specified filter. Filter is applied
431             * to all simons in the hierarchy of all types. If a simon is rejected by filter its children are not considered.
432             * Simons are aggregated in the top-bottom fashion, i.e. parent simons are aggregated before children.
433             *
434             * @param simon root of the hierarchy of simons for which statistics will be aggregated
435             * @param filter filter to select subsets of simons to aggregate
436             * @return aggregates statistics
437             * @since 3.5
438             */
439            public static StopwatchAggregate calculateStopwatchAggregate(Simon simon, SimonFilter filter) {
440                    StopwatchAggregate stopwatchAggregate = new StopwatchAggregate();
441                    if (filter != null) {
442                            aggregateStopwatches(stopwatchAggregate, simon, filter);
443                    } else {
444                            aggregateStopwatches(stopwatchAggregate, simon, ACCEPT_ALL_FILTER);
445                    }
446    
447                    return stopwatchAggregate;
448            }
449    
450            private static void aggregateStopwatches(StopwatchAggregate aggregate, Simon simon, SimonFilter filter) {
451                    if (filter.accept(simon)) {
452                            if (simon instanceof Stopwatch) {
453                                    Stopwatch stopwatch = (Stopwatch) simon;
454                                    aggregate.addSample(stopwatch.sample());
455                            }
456    
457                            for (Simon child : simon.getChildren()) {
458                                    aggregateStopwatches(aggregate, child, filter);
459                            }
460                    }
461            }

474            /**
475             * Aggregate statistics from all counters in hierarchy that pass specified filter. Filter is applied
476             * to all simons in the hierarchy of all types. If a simon is rejected by filter its children are not considered.
477             * Simons are aggregated in the top-bottom fashion, i.e. parent simons are aggregated before children.
478             *
479             * @param simon root of the hierarchy of simons for which statistics will be aggregated
480             * @param filter filter to select subsets of simons to aggregate
481             * @return aggregates statistics
482             * @since 3.5
483             */
484            public static CounterAggregate calculateCounterAggregate(Simon simon, SimonFilter filter) {
485                    CounterAggregate aggregate = new CounterAggregate();
486                    if (filter != null) {
487                            aggregateCounters(aggregate, simon, filter);
488                    } else {
489                            aggregateCounters(aggregate, simon, ACCEPT_ALL_FILTER);
490                    }
491                    return aggregate;
492            }
493    
494            private static void aggregateCounters(CounterAggregate aggregate, Simon simon, SimonFilter filter) {
495                    if (filter.accept(simon)) {
496                            if (simon instanceof Counter) {
497                                    Counter counter = (Counter) simon;
498                                    aggregate.addSample(counter.sample());
499                            }
500    
501                            for (Simon child : simon.getChildren()) {
502                                    aggregateCounters(aggregate, child, filter);
503                            }
504                    }
505            }

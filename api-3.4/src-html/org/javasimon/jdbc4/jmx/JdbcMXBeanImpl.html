001    package org.javasimon.jdbc4.jmx;
002    
003    import org.javasimon.*;
004    import org.javasimon.utils.SimonUtils;
005    
006    /**
007     * Jdbc MXBean implementation. See {@link JdbcMXBean} for interface
008     * documentation.
009     * Most methods are implemented by calling the {@link Manager}.
010     *
011     * @author Radovan Sninsky
012     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
013     * @since 2
014     */
015    public class JdbcMXBeanImpl implements JdbcMXBean {
016            private Manager manager;
017            private String prefix;
018    
019            /**
020             * MXBean constructor with prefix initialization to default ({@code org.javasimon.jdbc}).
021             *
022             * @param manager instance of {@link Manager}, typically {@code SimonManager.manager()}.
023             */
024            public JdbcMXBeanImpl(Manager manager) {
025                    this.manager = manager;
026                    this.prefix = "org.javasimon.jdbc";
027            }
028    
029            /**
030             * MXBean constructor with custom prefix initialization.
031             *
032             * @param manager instance of {@link Manager}, typically {@code SimonManager.manager()}.
033             * @param prefix  custom prefix
034             */
035            public JdbcMXBeanImpl(Manager manager, String prefix) {
036                    this.manager = manager;
037                    this.prefix = prefix;
038            }
039    
040            public String getPrefix() {
041                    return prefix;
042            }
043    
044            public void setPrefix(String value) {
045                    prefix = value;
046            }
047    
048            public void enableMonitoring() {
049                    Simon s = manager.getSimon(prefix);
050                    if (s != null) {
051                            s.setState(SimonState.ENABLED, true);
052                    }
053            }
054    
055            public void disableMonitoring() {
056                    Simon s = manager.getSimon(prefix);
057                    if (s != null) {
058                            s.setState(SimonState.DISABLED, true);
059                    }
060            }
061    
062            public boolean isMonitoringEnabled() {
063                    return manager.getSimon(prefix) != null && manager.getSimon(prefix).isEnabled();
064            }
065    
066            public JdbcObjectInfo connectionsStat() {
067                    Stopwatch s = manager.getStopwatch(prefix + ".conn");
068    
069                    if (s != null) {
070                            return new JdbcObjectInfo(
071                                    s.getActive(),
072                                    s.getMaxActive(),
073                                    s.getMaxActiveTimestamp(),
074                                    s.getCounter(),
075                                    s.getCounter() - s.getActive(),
076                                    s.getMin(),
077                                    s.getMax(),
078                                    s.getTotal()
079                            );
080                    }
081                    return null;
082            }
083    
084            public JdbcObjectInfo statementsStat() {
085                    Stopwatch s = manager.getStopwatch(prefix + ".stmt");
086    
087                    if (s != null) {
088                            return new JdbcObjectInfo(
089                                    s.getActive(),
090                                    s.getMaxActive(),
091                                    s.getMaxActiveTimestamp(),
092                                    s.getCounter(),
093                                    s.getCounter() - s.getActive(),
094                                    s.getMin(),
095                                    s.getMax(),
096                                    s.getTotal()
097                            );
098                    }
099                    return null;
100            }
101    
102            public JdbcObjectInfo resultsetsStat() {
103                    return null;
104            }
105    
106            public String[] getSqlCommands() {
107                    Simon s = manager.getSimon(prefix + ".sql");
108                    if (s != null) {
109                            String[] names = new String[s.getChildren().size()];
110                            int i = 0;
111                            for (Simon sn : s.getChildren()) {
112                                    names[i++] = SimonUtils.localName(sn.getName());
113                            }
114                            return names;
115                    } else {
116                            return new String[0];
117                    }
118            }
119    
120            public org.javasimon.jmx.StopwatchSample getSqlCommandStat(String cmdId) {
121                    if (manager.getSimon(prefix + ".sql." + cmdId) != null) {
122                            return new org.javasimon.jmx.StopwatchSample(
123                                    manager.getStopwatch(prefix + ".sql." + cmdId).sample());
124                    }
125                    return null;
126            }
127    
128            public String[] getSqls(String cmdId) {
129                    Simon s = manager.getSimon(prefix + ".sql." + cmdId);
130                    if (s != null) {
131                            String[] names = new String[s.getChildren().size()];
132                            int i = 0;
133                            for (Simon sn : s.getChildren()) {
134                                    names[i++] = SimonUtils.localName(sn.getName());
135                            }
136                            return names;
137                    } else {
138                            return new String[0];
139                    }
140            }
141    
142            public org.javasimon.jmx.StopwatchSample getSqlStat(String sqlId) {
143                    if (manager != null) {
144                            for (String simonName : manager.getSimonNames()) {
145                                    if (SimonUtils.localName(simonName).equals(sqlId)) {
146                                            return new org.javasimon.jmx.StopwatchSample(
147                                                    manager.getStopwatch(simonName).sample());
148                                    }
149                            }
150                    }
151                    return null;
152            }
153    }

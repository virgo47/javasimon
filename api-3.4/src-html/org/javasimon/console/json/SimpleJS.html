001    package org.javasimon.console.json;
002    
003    import java.io.IOException;
004    import java.io.Writer;
005    import org.javasimon.console.text.Stringifier;
006    
007    /**
008     * JavaScript simple type
009     *
010     * @author gquintana
011     */
012    public class SimpleJS<T> extends AnyJS {
013            /**
014             * Simple value
015             */
016            protected final T value;
017            /**
018             * Value formatter
019             */
020            protected final Stringifier<T> valueStringifier;
021            /**
022             * Hidden constructor use factory methods instead
023             */
024            public SimpleJS(T value, Stringifier<T> valueStringifier) {
025                    this.value = value;
026                    this.valueStringifier = valueStringifier;
027            }
028    
029            public T getValue() {
030                    return value;
031            }
032    
033            public String getFormattedValue() {
034                    return valueStringifier.toString(value);
035            }
036    
037            @Override
038            public void write(Writer writer) throws IOException {
039                    writer.write(getFormattedValue());
040            }
041    
042    }

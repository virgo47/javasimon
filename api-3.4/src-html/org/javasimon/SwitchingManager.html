001    package org.javasimon;
002    
003    import org.javasimon.callback.CompositeCallback;
004    
005    import java.util.Collection;
006    
007    /**
008     * {@link Manager} implementation that supports {@link #enable()} and {@link #disable()} and switches between
009     * backing {@link EnabledManager} and {@link DisabledManager} accordingly.
010     *
011     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
012     */
013    public final class SwitchingManager implements Manager {
014            private Manager enabled = new EnabledManager();
015    
016            private Manager disabled = new DisabledManager();
017    
018            private Manager manager = enabled;
019    
020            @Override
021            public Simon getRootSimon() {
022                    return manager.getRootSimon();
023            }
024    
025            @Override
026            public Simon getSimon(String name) {
027                    return manager.getSimon(name);
028            }
029    
030            @Override
031            public Counter getCounter(String name) {
032                    return manager.getCounter(name);
033            }
034    
035            @Override
036            public Stopwatch getStopwatch(String name) {
037                    return manager.getStopwatch(name);
038            }
039    
040            @Override
041            public Collection<String> getSimonNames() {
042                    return manager.getSimonNames();
043            }
044    
045            @Override
046            public Collection<Simon> getSimons(SimonPattern pattern) {
047                    return manager.getSimons(pattern);
048            }
049    
050            @Override
051            public void destroySimon(String name) {
052                    manager.destroySimon(name);
053            }
054    
055            @Override
056            public void clear() {
057                    manager.clear();
058            }
059    
060            @Override
061            public CompositeCallback callback() {
062                    return manager.callback();
063            }
064    
065            @Override
066            public ManagerConfiguration configuration() {
067                    return manager.configuration();
068            }
069    
070            /**
071             * Enables the Simon Manager. Enabled manager provides real Simons.
072             */
073            @Override
074            public void enable() {
075                    manager = enabled;
076            }
077    
078            /**
079             * Disables the Simon Manager. Disabled manager provides null Simons that actually do nothing.
080             */
081            @Override
082            public void disable() {
083                    manager = disabled;
084            }
085    
086            /**
087             * Returns true if the Java Simon API is enabled.
088             *
089             * @return true if the API is enabled
090             */
091            @Override
092            public boolean isEnabled() {
093                    return manager == enabled;
094            }
095    
096            @Override
097            public void message(String message) {
098                    manager.message(message);
099            }
100    
101            @Override
102            public void warning(String warning, Exception cause) {
103                    manager.warning(warning, cause);
104            }
105    }

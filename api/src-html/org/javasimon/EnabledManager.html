001    package org.javasimon;
002    
003    import java.lang.reflect.Constructor;
004    import java.lang.reflect.InvocationTargetException;
005    import java.util.ArrayList;
006    import java.util.Collection;
007    import java.util.Collections;
008    import java.util.Map;
009    import java.util.concurrent.ConcurrentHashMap;
010    
011    import org.javasimon.callback.CompositeCallback;
012    import org.javasimon.callback.CompositeCallbackImpl;
013    import org.javasimon.utils.SimonUtils;
014    
015    /**
016     * Implements fully functional {@link Manager} in the enabled state. Does not support
017     * {@link #enable()}/{@link #disable()} - for this use {@link SwitchingManager}.
018     *
019     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
020     */
021    public final class EnabledManager implements Manager {
022            private final Map<String, AbstractSimon> allSimons = new ConcurrentHashMap<String, AbstractSimon>();
023    
024            private UnknownSimon rootSimon;
025    
026            private CompositeCallback callback = new CompositeCallbackImpl();
027    
028            private ManagerConfiguration configuration;
029    
030            /**
031             * Creates new enabled manager.
032             */
033            public EnabledManager() {
034                    rootSimon = new UnknownSimon(ROOT_SIMON_NAME, this);
035                    allSimons.put(ROOT_SIMON_NAME, rootSimon);
036                    configuration = new ManagerConfiguration(this);
037                    callback.initialize();
038            }
039    
040            @Override
041            public Simon getSimon(String name) {
042                    return allSimons.get(name);
043            }
044    
045            @Override
046            public synchronized void destroySimon(String name) {
047                    if (name.equals(ROOT_SIMON_NAME)) {
048                            throw new SimonException("Root Simon cannot be destroyed!");
049                    }
050                    AbstractSimon simon = allSimons.remove(name);
051                    if (simon.getChildren().size() > 0) {
052                            replaceUnknownSimon(simon, UnknownSimon.class);
053                    } else {
054                            ((AbstractSimon) simon.getParent()).replaceChild(simon, null);
055                    }
056                    callback.onSimonDestroyed(simon);
057            }
058    
059            @Override
060            public synchronized void clear() {
061                    allSimons.clear();
062                    rootSimon = new UnknownSimon(ROOT_SIMON_NAME, this);
063                    allSimons.put(ROOT_SIMON_NAME, rootSimon);
064                    callback.onManagerClear();
065            }
066    
067            @Override
068            public Counter getCounter(String name) {
069                    return (Counter) getOrCreateSimon(name, CounterImpl.class);
070            }
071    
072            @Override
073            public Stopwatch getStopwatch(String name) {
074                    return (Stopwatch) getOrCreateSimon(name, StopwatchImpl.class);
075            }
076    
077            @Override
078            public Simon getRootSimon() {
079                    return rootSimon;
080            }
081    
082            @Override
083            public Collection<String> getSimonNames() {
084                    return Collections.unmodifiableCollection(allSimons.keySet());
085            }
086    
087            @SuppressWarnings({"unchecked"})
088            @Override
089            public Collection<Simon> getSimons(SimonPattern pattern) {
090                    if (pattern == null) {
091                            return Collections.unmodifiableCollection((Collection) allSimons.values());
092                    }
093                    Collection<Simon> simons = new ArrayList<Simon>();
094                    for (Map.Entry<String, AbstractSimon> entry : allSimons.entrySet()) {
095                            if (pattern.matches(entry.getKey())) {
096                                    simons.add(entry.getValue());
097                            }
098                    }
099                    return simons;
100            }
101    
102            private Simon getOrCreateSimon(String name, Class<? extends AbstractSimon> simonClass) {
103                    if (name == null) {
104                            // create an "anonymous" Simon - Manager does not care about it anymore
105                            return instantiateSimon(null, simonClass);
106                    }
107                    if (name.equals(ROOT_SIMON_NAME)) {
108                            throw new SimonException("Root Simon cannot be replaced or recreated!");
109                    }
110                    AbstractSimon simon = allSimons.get(name);
111                    if (simon != null && simonClass.isInstance(simon)) {
112                            return simon;
113                    }
114                    return createOrReplaceUnknownSimon(name, simonClass);
115            }
116    
117            private synchronized AbstractSimon createOrReplaceUnknownSimon(String name, Class<? extends AbstractSimon> simonClass) {
118                    // we will rather check the map in synchronized block before we try to create/replace the Simon
119                    AbstractSimon simon = allSimons.get(name);
120                    if (simon != null && simonClass.isInstance(simon)) {
121                            return simon; // the same return like in non-synchronized getOrCreateSimon - you just never know
122                    }
123    
124                    if (simon == null) {
125                            if (name != null && !SimonUtils.checkName(name)) {
126                                    throw new SimonException("Simon name must match following pattern: '" + SimonUtils.NAME_PATTERN.pattern() + "', used name: " + name);
127                            }
128                            simon = newSimon(name, simonClass);
129                    } else if (simon instanceof UnknownSimon) {
130                            simon = replaceUnknownSimon(simon, simonClass);
131                    } else {
132                            throw new SimonException("Simon named '" + name + "' already exists and its type is '" + simon.getClass().getName() + "' while requested type is '" + simonClass.getName() + "'.");
133                    }
134                    callback.onSimonCreated(simon);
135                    return simon;
136            }
137    
138            // called from synchronized method
139            private AbstractSimon replaceUnknownSimon(AbstractSimon simon, Class<? extends AbstractSimon> simonClass) {
140                    AbstractSimon newSimon = instantiateSimon(simon.getName(), simonClass);
141                    newSimon.enabled = simon.enabled;
142    
143                    // fixes parent link and parent's children list
144                    ((AbstractSimon) simon.getParent()).replaceChild(simon, newSimon);
145    
146                    // fixes children list and all children's parent link
147                    for (Simon child : simon.getChildren()) {
148                            newSimon.addChild((AbstractSimon) child);
149                            ((AbstractSimon) child).setParent(newSimon);
150                    }
151    
152                    allSimons.put(simon.getName(), newSimon);
153                    return newSimon;
154            }
155    
156            // called from synchronized method
157            private AbstractSimon newSimon(String name, Class<? extends AbstractSimon> simonClass) {
158                    AbstractSimon simon = instantiateSimon(name, simonClass);
159                    if (name != null) {
160                            addToHierarchy(simon, name);
161                            SimonConfiguration config = configuration.getConfig(name);
162                            if (config.getState() != null) {
163                                    simon.setState(config.getState(), false);
164                            }
165                            allSimons.put(name, simon);
166                    }
167                    return simon;
168            }
169    
170            private AbstractSimon instantiateSimon(String name, Class<? extends AbstractSimon> simonClass) {
171                    AbstractSimon simon;
172                    try {
173                            Constructor<? extends AbstractSimon> constructor = simonClass.getDeclaredConstructor(String.class, Manager.class);
174                            simon = constructor.newInstance(name, this);
175                    } catch (NoSuchMethodException e) {
176                            throw new SimonException(e);
177                    } catch (InvocationTargetException e) {
178                            throw new SimonException(e);
179                    } catch (IllegalAccessException e) {
180                            throw new SimonException(e);
181                    } catch (InstantiationException e) {
182                            throw new SimonException(e);
183                    }
184                    return simon;
185            }
186    
187            private void addToHierarchy(AbstractSimon simon, String name) {
188                    int ix = name.lastIndexOf(HIERARCHY_DELIMITER);
189                    AbstractSimon parent = rootSimon;
190                    if (ix != -1) {
191                            String parentName = name.substring(0, ix);
192                            parent = allSimons.get(parentName);
193                            if (parent == null) {
194                                    parent = new UnknownSimon(parentName, this);
195                                    addToHierarchy(parent, parentName);
196                                    allSimons.put(parentName, parent);
197                            }
198                    }
199                    parent.addChild(simon);
200            }
201    
202            @Override
203            public CompositeCallback callback() {
204                    return callback;
205            }
206    
207            @Override
208            public ManagerConfiguration configuration() {
209                    return configuration;
210            }
211    
212            /**
213             * Throws {@link UnsupportedOperationException}.
214             */
215            @Override
216            public void enable() {
217                    throw new UnsupportedOperationException("Only SwitchingManager supports this operation.");
218            }
219    
220            /**
221             * Throws {@link UnsupportedOperationException}.
222             */
223            @Override
224            public void disable() {
225                    throw new UnsupportedOperationException("Only SwitchingManager supports this operation.");
226            }
227    
228            /**
229             * Returns true.
230             *
231             * @return true
232             */
233            @Override
234            public boolean isEnabled() {
235                    return true;
236            }
237    
238            @Override
239            public void message(String message) {
240                    callback.onManagerMessage(message);
241            }
242    
243            @Override
244            public void warning(String warning, Exception cause) {
245                    callback.onManagerWarning(warning, cause);
246            }
247    }

001    package org.javasimon;
002    
003    import org.javasimon.utils.SimonUtils;
004    
005    /**
006     * Object holds all relevant data from Stopwatch Simon. Whenever it is important to get more values
007     * in a synchronous manner, {@link org.javasimon.Stopwatch#sample()} (or {@link Stopwatch#sampleAndReset()}
008     * should be used to obtain this Java Bean object.
009     *
010     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
011     */
012    public class StopwatchSample extends Sample {
013            private long total;
014            private long counter;
015            private long min;
016            private long max;
017            private long minTimestamp;
018            private long maxTimestamp;
019            private long active;
020            private long maxActive;
021            private long maxActiveTimestamp;
022            private long last;
023            private double mean;
024            private double standardDeviation;
025            private double variance;
026            private double varianceN;
027    
028            /**
029             * Returns the total sum of all split times in nanoseconds.
030             *
031             * @return total time of the stopwatch in nanoseconds
032             */
033            public final long getTotal() {
034                    return total;
035            }
036    
037            /**
038             * Sets the total sum of all split times in nanoseconds.
039             *
040             * @param total total time of the stopwatch in nanoseconds
041             */
042            public final void setTotal(long total) {
043                    this.total = total;
044            }
045    
046            /**
047             * Returns usage count of the stopwatch. Counter is increased by {@code addTime} and
048             * {@code stop} - that means that it's updated every time the next time split is added.
049             *
050             * @return count of time splits
051             */
052            public final long getCounter() {
053                    return counter;
054            }
055    
056            /**
057             * Sets the usage count of the stopwatch.
058             *
059             * @param counter count of time splits
060             */
061            public final void setCounter(long counter) {
062                    this.counter = counter;
063            }
064    
065            /**
066             * Returns minimal time split value in nanoseconds.
067             *
068             * @return minimal time split in nanoseconds
069             */
070            public final long getMin() {
071                    return min;
072            }
073    
074            /**
075             * Sets the minimal time split value in nanoseconds.
076             *
077             * @param min minimal time split in nanoseconds
078             */
079            public final void setMin(long min) {
080                    this.min = min;
081            }
082    
083            /**
084             * Returns maximal time split value in nanoseconds.
085             *
086             * @return maximal time split in nanoseconds
087             */
088            public final long getMax() {
089                    return max;
090            }
091    
092            /**
093             * Sets the maximal time split value in nanoseconds.
094             *
095             * @param max maximal time split in nanoseconds
096             */
097            public final void setMax(long max) {
098                    this.max = max;
099            }
100    
101            /**
102             * Returns ms timestamp when the min value was measured.
103             *
104             * @return ms timestamp of the min value measurement
105             */
106            public final long getMinTimestamp() {
107                    return minTimestamp;
108            }
109    
110            /**
111             * Sets the ms timestamp when the min value was measured.
112             *
113             * @param minTimestamp ms timestamp of the min value measurement
114             */
115            public final void setMinTimestamp(long minTimestamp) {
116                    this.minTimestamp = minTimestamp;
117            }
118    
119            /**
120             * Returns ms timestamp when the max value was measured.
121             *
122             * @return ms timestamp of the max value measurement
123             */
124            public final long getMaxTimestamp() {
125                    return maxTimestamp;
126            }
127    
128            /**
129             * Sets the ms timestamp when the max value was measured.
130             *
131             * @param maxTimestamp ms timestamp of the max value measurement
132             */
133            public final void setMaxTimestamp(long maxTimestamp) {
134                    this.maxTimestamp = maxTimestamp;
135            }
136    
137            /**
138             * Returns current number of measured splits (concurrently running).
139             *
140             * @return current number of active splits
141             */
142            public final long getActive() {
143                    return active;
144            }
145    
146            /**
147             * Sets the current number of measured splits (concurrently running).
148             *
149             * @param active current number of active splits
150             */
151            public final void setActive(long active) {
152                    this.active = active;
153            }
154    
155            /**
156             * Returns peek value of active concurrent splits.
157             *
158             * @return maximum reached value of active splits
159             */
160            public final long getMaxActive() {
161                    return maxActive;
162            }
163    
164            /**
165             * Sets the peek value of active concurrent splits.
166             *
167             * @param maxActive maximum reached value of active splits
168             */
169            public final void setMaxActive(long maxActive) {
170                    this.maxActive = maxActive;
171            }
172    
173            /**
174             * Returns ms timestamp when the last peek of the active split count occured.
175             *
176             * @return ms timestamp of the last peek of the active split count
177             */
178            public final long getMaxActiveTimestamp() {
179                    return maxActiveTimestamp;
180            }
181    
182            /**
183             * Sets the ms timestamp when the last peek of the active split count occured.
184             *
185             * @param maxActiveTimestamp ms timestamp of the last peek of the active split count
186             */
187            public final void setMaxActiveTimestamp(long maxActiveTimestamp) {
188                    this.maxActiveTimestamp = maxActiveTimestamp;
189            }
190    
191            /**
192             * Returns the value of the last measured split in ns.
193             *
194             * @return last measured split in ns
195             */
196            public final long getLast() {
197                    return last;
198            }
199    
200            /**
201             * Sets the value of the last measured split in ns.
202             *
203             * @param last last measured split in ns
204             */
205            public final void setLast(long last) {
206                    this.last = last;
207            }
208    
209            /**
210             * Returns mean value (average) of all measured values.
211             *
212             * @return mean value
213             */
214            public final double getMean() {
215                    return mean;
216            }
217    
218            /**
219             * Sets the mean value (average) of all measured values.
220             *
221             * @param mean mean value
222             */
223            public final void setMean(double mean) {
224                    this.mean = mean;
225            }
226    
227            /**
228             * Returns standard deviation for all measured values.
229             *
230             * @return standard deviation
231             */
232            public final double getStandardDeviation() {
233                    return standardDeviation;
234            }
235    
236            /**
237             * Sets the standard deviation for all measured values.
238             *
239             * @param standardDeviation standard deviation
240             */
241            public final void setStandardDeviation(double standardDeviation) {
242                    this.standardDeviation = standardDeviation;
243            }
244    
245            /**
246             * Returns unbiased estimate of the population variance.
247             *
248             * @return unbiased estimated variance
249             */
250            public final double getVariance() {
251                    return variance;
252            }
253    
254            /**
255             * Sets the unbiased estimate of the population variance.
256             *
257             * @param variance unbiased estimated variance
258             */
259            public final void setVariance(double variance) {
260                    this.variance = variance;
261            }
262    
263            /**
264             * Returns variance value of all measured values (entire population).
265             *
266             * @return entire population variance
267             */
268            public final double getVarianceN() {
269                    return varianceN;
270            }
271    
272            /**
273             * Sets the variance value of all measured values (entire population).
274             *
275             * @param varianceN entire population variance
276             */
277            public final void setVarianceN(double varianceN) {
278                    this.varianceN = varianceN;
279            }
280    
281            /**
282             * Returns readable representation of object.
283             *
284             * @return string with readable representation of object
285             */
286            @Override
287            public String toString() {
288                    final StringBuilder sb = new StringBuilder();
289                    sb.append("StopwatchSample");
290                    sb.append("{name=").append(getName());
291                    sb.append(", total=").append(SimonUtils.presentNanoTime(total));
292                    sb.append(", counter=").append(counter);
293                    sb.append(", min=").append(SimonUtils.presentMinMaxSplit(min));
294                    sb.append(", max=").append(SimonUtils.presentMinMaxSplit(max));
295                    sb.append(", minTimestamp=").append(SimonUtils.presentTimestamp(minTimestamp));
296                    sb.append(", maxTimestamp=").append(SimonUtils.presentTimestamp(maxTimestamp));
297                    sb.append(", active=").append(active);
298                    sb.append(", maxActive=").append(maxActive);
299                    sb.append(", maxActiveTimestamp=").append(SimonUtils.presentTimestamp(maxActiveTimestamp));
300                    sb.append(", last=").append(SimonUtils.presentNanoTime(last));
301                    sb.append(", mean=").append(SimonUtils.presentNanoTime((long) getMean()));
302                    sb.append(", standardDeviation=").append(SimonUtils.presentNanoTime((long) getStandardDeviation()));
303                    sb.append(", variance=").append(getVariance());
304                    sb.append(", varianceN=").append(getVarianceN());
305                    finishWithUsagesAndNote(sb);
306                    return sb.toString();
307            }
308    
309            /**
310             * Equivalent to {@link org.javasimon.StopwatchImpl#toString()} without state.
311             */
312            public String stopwatchToString() {
313                    return "Simon Stopwatch: total " + SimonUtils.presentNanoTime(total) +
314                            ", counter " + counter +
315                            ", max " + SimonUtils.presentNanoTime(max) +
316                            ", min " + SimonUtils.presentNanoTime(min) +
317                            ", mean " + SimonUtils.presentNanoTime((long) mean) +
318                            simonToString();
319            }
320    }
